    <div id="ui">
        LOOP: <span id="loop-ui">1</span>/5 | 
        TIME REMAINING: <span id="time-ui">10.00</span>s | 
        DATA SHARDS: <span id="shard-ui">0/3</span>
    </div>

    <div id="canvas-container">
        <canvas id="gameCanvas" width="600" height="600"></canvas>
        <div id="overlay">
            <h1 id="title-ui" style="color: #ff3366; font-size: 36px; margin: 0; text-shadow: 0 0 12px #ff3366;">PARADOX_DETECTED</h1>
            <p id="msg-ui" style="font-size: 16px; margin-top: 10px; text-align: center; max-width: 80%;">You collided with a past iteration of yourself.</p>
            <button id="action-btn" onclick="resetGame()">RE-INITIALIZE TIMELINE</button>
        </div>
    </div>

    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        const loopUi = document.getElementById('loop-ui');
        const timeUi = document.getElementById('time-ui');
        const shardUi = document.getElementById('shard-ui');
        const overlay = document.getElementById('overlay');
        const titleUi = document.getElementById('title-ui');
        const msgUi = document.getElementById('msg-ui');
        const actionBtn = document.getElementById('action-btn');

        // Game Matrix Configuration
        const LOOP_DURATION = 10; 
        const MAX_LOOPS = 5;
        const TARGET_SHARDS = 3;

        let currentLoop = 1;
        let timeElapsed = 0;
        let isGameOver = false;
        let lastTime = 0;

        // Player Entity Structure
        let player = { x: 300, y: 500, radius: 10, speed: 200 };
        let currentHistory = []; 
        let pastHistories = [];   

        // Environment Obstacles & Items
        let shards = [];
        let drones = [
            { x: 150, y: 180, vx: 160, vy: 0, radius: 12 },
            { x: 450, y: 320, vx: -160, vy: 0, radius: 12 },
            { x: 300, y: 120, vx: 0, vy: 140, radius: 12 }
        ];

        // Key Input Listeners
        let keys = {};
        window.addEventListener('keydown', e => keys[e.key.toLowerCase()] = true);
        window.addEventListener('keyup', e => keys[e.key.toLowerCase()] = false);

        function spawnShards() {
            shards = [];
            const structuralPositions = [
                {x: 120, y: 250}, {x: 480, y: 250}, {x: 300, y: 220}, 
                {x: 150, y: 420}, {x: 450, y: 420}, {x: 300, y: 50}
            ];
            let shuffled = structuralPositions.sort(() => 0.5 - Math.random());
            for(let i = 0; i < TARGET_SHARDS; i++) {
                shards.push({ x: shuffled[i].x, y: shuffled[i].y, radius: 7, collected: false });
            }
        }

        function resetGame() {
            currentLoop = 1; 
            timeElapsed = 0; 
            pastHistories = []; 
            currentHistory = [];
            player.x = 300; 
            player.y = 500; 
            isGameOver = false;
            overlay.style.display = 'none';
            spawnShards();
            lastTime = performance.now();
            requestAnimationFrame(gameLoop);
        }

        function terminateSimulation(victory, feedbackMessage) {
            isGameOver = true;
            overlay.style.display = 'flex';
            if (victory) {
                titleUi.textContent = "CORE_DECRYPTED";
                titleUi.style.color = "#33ff33";
                titleUi.style.textShadow = "0 0 12px #33ff33";
                msgUi.textContent = "Data integrity secure. All architectural cycles verified successfully.";
                actionBtn.textContent = "RE-EXECUTE SYSTEM";
            } else {
                titleUi.textContent = "TIMELINE_COLLAPSED";
                titleUi.style.color = "#ff3366";
                titleUi.style.textShadow = "0 0 12px #ff3366";
                msgUi.textContent = feedbackMessage;
                actionBtn.textContent = "RE-INITIALIZE TIMELINE";
            }
        }

        function gameLoop(timestamp) {
            if (isGameOver) return;
            
            let dt = (timestamp - lastTime) / 1000;
            if (dt > 0.1) dt = 0.1; // Execution safety clamp
            lastTime = timestamp;

            timeElapsed += dt;
            let currentRemaining = Math.max(0, LOOP_DURATION - timeElapsed);

            // Vector input normalization math
            let moveX = 0, moveY = 0;
            if (keys['w'] || keys['arrowup']) moveY -= 1;
            if (keys['s'] || keys['arrowdown']) moveY += 1;
            if (keys['a'] || keys['arrowleft']) moveX -= 1;
            if (keys['d'] || keys['arrowright']) moveX += 1;

            if (moveX !== 0 && moveY !== 0) { 
                moveX *= 0.7071; 
                moveY *= 0.7071;
            }

            player.x += moveX * player.speed * dt;
            player.y += moveY * player.speed * dt;

            // Strict Boundary Intersections
            player.x = Math.max(player.radius, Math.min(600 - player.radius, player.x));
            player.y = Math.max(player.radius, Math.min(600 - player.radius, player.y));

            // Record active frame location state
            currentHistory.push({ x: player.x, y: player.y });

            // Process Autonomous Drones Mechanics
            drones.forEach(d => {
                d.x += d.vx * dt; 
                d.y += d.vy * dt;
                if (d.x <= d.radius || d.x >= 600 - d.radius) d.vx *= -1;
                if (d.y <= d.radius || d.y >= 600 - d.radius) d.vy *= -1;

                if (Math.hypot(player.x - d.x, player.y - d.y) < player.radius + d.radius) {
                    terminateSimulation(false, "Vaporized by active firewall network node.");
                }
            });

            // Shard collection intersection tracking
            shards.forEach(s => {
                if (!s.collected && Math.hypot(player.x - s.x, player.y - s.y) < player.radius + s.radius) {
                    s.collected = true;
                }
            });

            // Process History Playback Loops and Paradox Detections
            let currentPlaybackFrame = currentHistory.length - 1;
            for (let i = 0; i < pastHistories.length; i++) {
                let historicalRun = pastHistories[i];
                let pastFrame = historicalRun[Math.min(currentPlaybackFrame, historicalRun.length - 1)];
                if (pastFrame) {
                    let spatialDistance = Math.hypot(player.x - pastFrame.x, player.y - pastFrame.y);
                    if (spatialDistance < player.radius * 2) {
                        terminateSimulation(false, `Temporal Paradox! Intersected path with Echo Instance #${i + 1}.`);
                        return;
                    }
                }
            }

            // --- Graphics Render Pass ---
            ctx.clearRect(0, 0, 600, 600);

            // Vector Geometry Grid
            ctx.strokeStyle = '#121224'; 
            ctx.lineWidth = 1;
            for(let i = 0; i < 600; i += 40) {
                ctx.beginPath(); ctx.moveTo(i, 0); ctx.lineTo(i, 600); ctx.stroke();
                ctx.beginPath(); ctx.moveTo(0, i); ctx.lineTo(600, i); ctx.stroke();
            }

            // Draw Data Targets
            shards.forEach(s => {
                if (!s.collected) {
                    ctx.beginPath(); ctx.arc(s.x, s.y, s.radius, 0, Math.PI*2);
                    ctx.fillStyle = '#ffff33'; ctx.shadowBlur = 10; ctx.shadowColor = '#ffff33';
                    ctx.fill();
                }
            });
            ctx.shadowBlur = 0; 

            // Render Historical Playback Channels (Echo Clones)
            pastHistories.forEach((historicalRun) => {
                let pastFrame = historicalRun[Math.min(currentPlaybackFrame, historicalRun.length - 1)];
                if (pastFrame) {
                    ctx.beginPath(); ctx.arc(pastFrame.x, pastFrame.y, player.radius, 0, Math.PI*2);
                    ctx.fillStyle = 'rgba(0, 255, 255, 0.25)';
                    ctx.strokeStyle = '#00ffff'; ctx.lineWidth = 1.5;
                    ctx.fill(); ctx.stroke();
                }
            });

            // Render Security Drones
            drones.forEach(d => {
                ctx.beginPath(); ctx.arc(d.x, d.y, d.radius, 0, Math.PI*2);
                ctx.fillStyle = '#ff3366'; ctx.shadowBlur = 8; ctx.shadowColor = '#ff3366';
                ctx.fill();
            });
            ctx.shadowBlur = 0;

            // Render Active Controlled Character Instance
            ctx.beginPath(); ctx.arc(player.x, player.y, player.radius, 0, Math.PI*2);
            ctx.fillStyle = '#33ff33'; ctx.shadowBlur = 12; ctx.shadowColor = '#33ff33';
            ctx.fill(); 
            ctx.shadowBlur = 0;

            // Sync HTML text components
            let collectedCount = shards.filter(s => s.collected).length;
            loopUi.textContent = currentLoop;
            timeUi.textContent = currentRemaining.toFixed(2);
            shardUi.textContent = `${collectedCount}/${TARGET_SHARDS}`;

            // Check Matrix State Completion Targets
            if (timeElapsed >= LOOP_DURATION) {
                if (collectedCount < TARGET_SHARDS) {
                    terminateSimulation(false, "Chronology buffer exhausted before data extraction was completed.");
                } else if (currentLoop >= MAX_LOOPS) {
                    terminateSimulation(true);
                } else {
                    pastHistories.push([...currentHistory]);
                    currentHistory = [];
                    currentLoop++;
                    timeElapsed = 0;
                    player.x = 300; 
                    player.y = 500; 
                    spawnShards();
                }
            }

            requestAnimationFrame(gameLoop);
        }

        // Initialize Simulation Engine
        spawnShards();
        lastTime = performance.now();
        requestAnimationFrame(gameLoop);
    </script>
